                    const type = target.dataset.type;
                    if (!type) return;

                    if (currentIndex < questions.length) {
                        scores[type] += 1;

                        if (currentIndex === questions.length - 1) {
                            currentIndex = 20;
                            render();
                        } else {
                            currentIndex++;
                            render();
                        }
                    }
                }
            }

            app.addEventListener('click', handleClick);
            render();
        })();
    </script>
